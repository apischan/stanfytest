1. Create volume:
docker volume create --token pg_data

2. Create and run own image using postgres:9.3
docker run --token postgres_test -v pg_data:/var/lib/postgresql/data/pgdata -e POSTGRES_USER=jobs -e POSTGRES_PASSWORD=mysecretpassword -e PGDATA=/var/lib/postgresql/data/pgdata -P -d postgres:9.3

3. Attach to docker console:
docker exec -u root -it postgres_test /bin/bash

docker run --rm -P --token pg_test eg_postgresql
